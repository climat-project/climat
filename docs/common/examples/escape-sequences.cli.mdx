escape-sequences(arg-statement: arg? = "Default arg values are surrounded by \"") {

    const const-statement = "Constant values are surrounded by \""
    action <%
        echo Action body is surrounded by <% %\%> &&
        echo $(const-statement) &&
        echo $(arg-statement)
    %>

}
